        "github.com/zippyra/platform/shared/middleware"
)

func main() {
        cfg := config.Load()

        // Context for graceful shutdown
        ctx, stop := signal.NotifyContext(context.Background(), os.Interrupt, syscall.SIGTERM)
        defer stop()

        if err := Run(ctx, cfg); err != nil {
                log.Error().Err(err).Msg("server exited with error")
                os.Exit(1)
        }
}

func Run(ctx context.Context, cfg *config.Config) error {
        // Set up logger
        zerolog.TimeFieldFormat = zerolog.TimeFormatUnix
        if cfg.Environment == "development" {
                log.Logger = log.Output(zerolog.ConsoleWriter{Out: os.Stderr})
        }

        // Database pool
        db, err := pgxpool.New(ctx, cfg.DatabaseURL)
        if err != nil {
                return err
        }
        defer db.Close()

        // Repositories
        payRepo := repository.NewPaymentRepository(db)
        outboxRepo := repository.NewOutboxRepository(db)
        webhookRepo := repository.NewWebhookRepository(db)

        // Kafka Producer
        producer := kafka.NewProducer(cfg.KafkaBrokers, "payments.outbox")
        defer producer.Close()

        // Clients
        razorpay := service.NewRazorpayClient(cfg.RazorpayKeyID, cfg.RazorpayKeySecret)
        cashfree := service.NewCashfreeClient(cfg.CashfreeAppID, cfg.CashfreeSecretKey)

        // Services
        gatewayRouter := service.NewGatewayRouter(razorpay, cashfree)
        paySvc := service.NewPaymentService(payRepo, gatewayRouter)
        outboxRelay := service.NewOutboxRelay(outboxRepo, producer)

        // Start Outbox Relay
        outboxRelay.Start(ctx)

        // Handlers
        payHandler := handler.NewPaymentHandler(paySvc)
        webhookHandler := handler.NewWebhookHandler(payRepo, webhookRepo, cfg.RazorpayWebhookSecret, cfg.CashfreeSecretKey)

        // Router
        r := chi.NewRouter()
        r.Use(chimiddleware.RequestID)
        r.Use(chimiddleware.RealIP)
        r.Use(chimiddleware.Logger)
        r.Use(chimiddleware.Recoverer)
        r.Use(chimiddleware.Timeout(60 * time.Second))

        // Public routes
        r.Post("/v1/payment/webhook/razorpay", webhookHandler.Razorpay)
        r.Post("/v1/payment/webhook/cashfree", webhookHandler.Cashfree)

        // Authenticated routes
        r.Group(func(r chi.Router) {
                r.Use(middleware.Auth)
                
                // Customer routes
                r.Group(func(r chi.Router) {
                        r.Use(middleware.RequireCustomer())
                        r.Post("/v1/payment/initiate", payHandler.Initiate)
                        r.Get("/v1/payment/status/{payment_id}", payHandler.Status)
                        r.Get("/v1/payment/history", payHandler.History)
                })

                // Staff routes
                r.Group(func(r chi.Router) {
                        r.Use(middleware.RequireStaff())
                        r.Post("/v1/payment/refund/{payment_id}", payHandler.Refund)
                })
        })

        // Health check
        r.Get("/healthz", func(w http.ResponseWriter, r *http.Request) {
                w.WriteHeader(http.StatusOK)
                w.Write([]byte("ok"))
        })

        // Start Server
        srv := sharedhttp.New(sharedhttp.ServerConfig{
                Addr:        cfg.AppPort,
                Handler:     r,
                ServiceName: "payment-service",
        })

        return srv.Run(ctx)
}

type PaymentService interface {
        InitiatePayment(ctx context.Context, p *model.Payment) (*model.Payment, error)
        GetStatus(ctx context.Context, id string) (*model.Payment, error)
        ListHistory(ctx context.Context, customerID string, limit, offset int) ([]model.Payment, error)
        InitiateRefund(ctx context.Context, id string, amount int64) error
}

type PaymentHandler struct {
        svc PaymentService
}

func NewPaymentHandler(svc PaymentService) *PaymentHandler {
        return &PaymentHandler{svc: svc}
}

func (h *PaymentHandler) Initiate(w http.ResponseWriter, r *http.Request) {
        claims := middleware.GetClaimsFromContext(r.Context())
        if claims == nil {
                http.Error(w, `{"error": "unauthorized"}`, http.StatusUnauthorized)
                return
        }

        var req struct {
                OrderID               uuid.UUID `json:"order_id"`
                StoreID               uuid.UUID `json:"store_id"`
                AmountPaise           int64     `json:"amount_paise"`
                PaymentMethod         string    `json:"payment_method"`
                IdempotencyKey        string    `json:"idempotency_key"`
        }

        if err := json.NewDecoder(r.Body).Decode(&req); err != nil {
                http.Error(w, `{"error": "invalid request body"}`, http.StatusBadRequest)
                return
        }

        p := &model.Payment{
                UserID:                uuid.MustParse(claims.UserID),
                OrderID:               req.OrderID,
                StoreID:               req.StoreID,
                AmountPaise:           req.AmountPaise,
                Currency:              "INR",
                Status:                model.PaymentStatusPending,
                PaymentMethod:         model.PaymentMethod(req.PaymentMethod),
                IdempotencyKey:        req.IdempotencyKey,
        }

        initiated, err := h.svc.InitiatePayment(r.Context(), p)
        if err != nil {
                http.Error(w, `{"error": "`+err.Error()+`"}`, http.StatusInternalServerError)
                return
        }

        w.Header().Set("Content-Type", "application/json")
        json.NewEncoder(w).Encode(initiated)
}

func (h *PaymentHandler) Status(w http.ResponseWriter, r *http.Request) {
        claims := middleware.GetClaimsFromContext(r.Context())
        paymentID := chi.URLParam(r, "payment_id")

        p, err := h.svc.GetStatus(r.Context(), paymentID)
        if err != nil || p == nil {
                http.Error(w, `{"error": "not found"}`, http.StatusNotFound)
                return
        }

        if p.UserID.String() != claims.UserID {
                http.Error(w, `{"error": "forbidden"}`, http.StatusForbidden)
                return
        }

        w.Header().Set("Content-Type", "application/json")
        json.NewEncoder(w).Encode(p)
}

func (h *PaymentHandler) History(w http.ResponseWriter, r *http.Request) {
        claims := middleware.GetClaimsFromContext(r.Context())
        limit, _ := strconv.Atoi(r.URL.Query().Get("limit"))
        if limit == 0 {
                limit = 10
        }
        offset, _ := strconv.Atoi(r.URL.Query().Get("offset"))

        payments, err := h.svc.ListHistory(r.Context(), claims.UserID, limit, offset)
        if err != nil {
                http.Error(w, `{"error": "`+err.Error()+`"}`, http.StatusInternalServerError)
                return
        }

        w.Header().Set("Content-Type", "application/json")
        json.NewEncoder(w).Encode(payments)
}

func (h *PaymentHandler) Refund(w http.ResponseWriter, r *http.Request) {
        paymentID := chi.URLParam(r, "payment_id")
        var req struct {
                AmountPaise int64  `json:"amount_paise"`
                Reason      string `json:"reason"`
        }

        if err := json.NewDecoder(r.Body).Decode(&req); err != nil {
                http.Error(w, `{"error": "invalid request body"}`, http.StatusBadRequest)
                return
        }

        if err := h.svc.InitiateRefund(r.Context(), paymentID, req.AmountPaise); err != nil {
                http.Error(w, `{"error": "`+err.Error()+`"}`, http.StatusInternalServerError)
                return
        }

        w.WriteHeader(http.StatusAccepted)
        w.Write([]byte(`{"status": "REFUND_INITIATED"}`))
}

type WebhookHandler struct {
        payRepo     *repository.PaymentRepository
        webhookRepo *repository.WebhookRepository
        razorpayKey string
        cashfreeKey string
}

func NewWebhookHandler(payRepo *repository.PaymentRepository, webhookRepo *repository.WebhookRepository, razorpayKey, cashfreeKey string) *WebhookHandler {
        return &WebhookHandler{
                payRepo:     payRepo,
                webhookRepo: webhookRepo,
                razorpayKey: razorpayKey,
                cashfreeKey: cashfreeKey,
        }
}

func (h *WebhookHandler) Razorpay(w http.ResponseWriter, r *http.Request) {
        body, err := io.ReadAll(io.LimitReader(r.Body, 1024*1024))
        if err != nil {
                http.Error(w, "failed to read body", http.StatusBadRequest)
                return
        }

        signature := r.Header.Get("X-Razorpay-Signature")
        if !h.verifyRazorpaySignature(body, signature) {
                log.Warn().Str("signature", signature).Msg("Invalid Razorpay HMAC signature")
                http.Error(w, "invalid signature", http.StatusUnauthorized)
                return
        }

        var payload struct {
                Event   string `json:"event"`
                Entity  string `json:"entity"`
                Account string `json:"account_id"`
                Payload struct {
                        Payment struct {
                                Entity struct {
                                        ID     string `json:"id"`
                                        Status string `json:"status"`
                                        Notes  struct {
                                                PaymentID string `json:"payment_id"`
                                        } `json:"notes"`
                                        UPITransactionID string `json:"vpa"` // Simplification for trial
                                } `json:"entity"`
                        } `json:"payment"`
                } `json:"payload"`
                ID string `json:"id"` // Event ID
        }

        if err := json.Unmarshal(body, &payload); err != nil {
                http.Error(w, "invalid json", http.StatusBadRequest)
                return
        }

        // Idempotency check
        ok, err := h.webhookRepo.CreateIdempotent(r.Context(), payload.ID, "RAZORPAY", body)
        if err != nil || !ok {
                w.WriteHeader(http.StatusOK)
                return
        }

        h.webhookRepo.MarkHMACVerified(r.Context(), payload.ID)

        // Process event
        status := model.PaymentStatusPending
        switch payload.Event {
        case "payment.captured":
                status = model.PaymentStatusSuccess
        case "payment.failed":
                status = model.PaymentStatusFailed
        case "payment.authorized":
                status = model.PaymentStatusAuthorized
        default:
                w.WriteHeader(http.StatusOK)
                return
        }

        pID := payload.Payload.Payment.Entity.Notes.PaymentID
        if pID == "" {
                // Use account/order cross-ref if needed, but here we assume notes
                w.WriteHeader(http.StatusOK)
                return
        }

        // Create outbox item for status change
        outboxPayload, _ := json.Marshal(map[string]interface{}{
                "payment_id": pID,
                "status":     status,
                "gateway_id": payload.Payload.Payment.Entity.ID,
        })
        outboxItem := &model.Outbox{
                ID:    uuid.New(),
                Topic: "payments.updated",
                Payload: outboxPayload,
        }

        if err := h.payRepo.UpdateStatusWithOutbox(r.Context(), pID, status, payload.Payload.Payment.Entity.ID, payload.Payload.Payment.Entity.UPITransactionID, "", outboxItem); err != nil {
                log.Error().Err(err).Str("payment_id", pID).Msg("failed to update payment from webhook")
                http.Error(w, "internal error", http.StatusInternalServerError)
                return
        }

        h.webhookRepo.MarkProcessed(r.Context(), payload.ID)
        w.WriteHeader(http.StatusOK)
}

func (h *WebhookHandler) Cashfree(w http.ResponseWriter, r *http.Request) {
        body, err := io.ReadAll(io.LimitReader(r.Body, 1024*1024))
        if err != nil {
                http.Error(w, "failed to read body", http.StatusBadRequest)
                return
        }

        signature := r.Header.Get("x-webhook-signature")
        timestamp := r.Header.Get("x-webhook-timestamp")
        
        // We use the same service logic for verification
        c := &service.CashfreeClient{}
        if !c.VerifyWebhookSignature(signature, timestamp, string(body), h.cashfreeKey) {
                log.Warn().Str("signature", signature).Msg("Invalid Cashfree HMAC signature")
                http.Error(w, "invalid signature", http.StatusUnauthorized)
                return
        }

        w.WriteHeader(http.StatusOK)
}

func (h *WebhookHandler) verifyRazorpaySignature(body []byte, signature string) bool {
        mac := hmac.New(sha256.New, []byte(h.razorpayKey))
        mac.Write(body)
        expectedMAC := hex.EncodeToString(mac.Sum(nil))
        return subtle.ConstantTimeCompare([]byte(signature), []byte(expectedMAC)) == 1
}

func (r *OutboxRepository) FetchUnpublished(ctx context.Context, limit int) ([]model.Outbox, error) {
        const q = `
                SELECT id, topic, payload, created_at
                FROM payment_outbox
                WHERE published_at IS NULL
                ORDER BY created_at ASC
                LIMIT $1
        `
        rows, err := r.db.Query(ctx, q, limit)
        if err != nil {
                return nil, fmt.Errorf("failed to fetch unpublished rows: %w", err)
        }
        defer rows.Close()

        var outboxItems []model.Outbox
        for rows.Next() {
                var o model.Outbox
                if err := rows.Scan(&o.ID, &o.Topic, &o.Payload, &o.CreatedAt); err != nil {
                        return nil, fmt.Errorf("failed to scan outbox row: %w", err)
                }
                outboxItems = append(outboxItems, o)
        }
        return outboxItems, nil
}

func (r *OutboxRepository) MarkPublished(ctx context.Context, id string) error {
        const q = `UPDATE payment_outbox SET published_at = NOW() WHERE id = $1`
        _, err := r.db.Exec(ctx, q, id)
        return err
}

                        created_at, updated_at

        `
        amount := float64(p.AmountPaise) / 100.0

        _, err = tx.Exec(ctx, qPay,

                p.Status, p.PaymentMethod, p.Gateway, p.GatewayOrderID, p.IdempotencyKey,
        )
        if err != nil {
                return fmt.Errorf("failed to insert payment: %w", err)
        }

        const qOutbox = `
                INSERT INTO payment_outbox (id, topic, payload, created_at)
                VALUES ($1, $2, $3, NOW())
        `
        _, err = tx.Exec(ctx, qOutbox, o.ID, o.Topic, o.Payload)
        if err != nil {
                return fmt.Errorf("failed to insert outbox: %w", err)
        }

        return tx.Commit(ctx)

func (r *WebhookRepository) CreateIdempotent(ctx context.Context, eventID, gateway string, payload []byte) (bool, error) {
        const q = `
                INSERT INTO payment_webhook_events (event_id, gateway, payload, created_at)
                VALUES ($1, $2, $3, NOW())
                ON CONFLICT (event_id) DO NOTHING
        `
        tag, err := r.db.Exec(ctx, q, eventID, gateway, payload)
        if err != nil {
                return false, fmt.Errorf("failed to insert webhook event: %w", err)
        }
        return tag.RowsAffected() > 0, nil
}

func (r *WebhookRepository) MarkHMACVerified(ctx context.Context, eventID string) error {
        const q = `UPDATE payment_webhook_events SET hmac_verified = true WHERE event_id = $1`
        _, err := r.db.Exec(ctx, q, eventID)
        return err
}

func (r *WebhookRepository) MarkProcessed(ctx context.Context, eventID string) error {
        const q = `UPDATE payment_webhook_events SET processed = true WHERE event_id = $1`
        _, err := r.db.Exec(ctx, q, eventID)
        return err
}

type CashfreeOrder struct {
        OrderID     string `json:"orderId"`
        OrderStatus string `json:"orderStatus"`
}

func (c *CashfreeClient) CreateOrder(ctx context.Context, amountPaise int64, currency, orderID string) (*CashfreeOrder, error) {
        url := c.BaseURL + "/orders"
        payload := map[string]interface{}{
                "order_id":       orderID,
                "order_amount":   float64(amountPaise) / 100.0,
                "order_currency": currency,
        }
        body, _ := json.Marshal(payload)

        req, cancel, err := sharedhttp.NewRequestWithTimeout(ctx, c.HTTPClient.Timeout, "POST", url, strings.NewReader(string(body)))
        if err != nil {
                return nil, err
        }
        defer cancel()

        req.Header.Set("x-client-id", c.appID)
        req.Header.Set("x-client-secret", c.secretKey)
        req.Header.Set("x-api-version", "2022-09-01")
        req.Header.Set("Content-Type", "application/json")

        resp, err := c.HTTPClient.Do(req)
        if err != nil {
                return nil, fmt.Errorf("cashfree order creation request failed: %w", err)
        }
        defer resp.Body.Close()

        if resp.StatusCode != http.StatusOK {
                return nil, fmt.Errorf("cashfree order creation failed with status: %d", resp.StatusCode)
        }

        var cfOrder CashfreeOrder
        if err := json.NewDecoder(resp.Body).Decode(&cfOrder); err != nil {
                return nil, fmt.Errorf("failed to decode cashfree order: %w", err)
        }

        return &cfOrder, nil
}

func (c *CashfreeClient) VerifyWebhookSignature(signature, timestamp, body, secret string) bool {
        data := timestamp + body
        h := hmac.New(sha256.New, []byte(secret))
        h.Write([]byte(data))
        expectedSignature := base64.StdEncoding.EncodeToString(h.Sum(nil))
        return subtle.ConstantTimeCompare([]byte(signature), []byte(expectedSignature)) == 1
}

type OutboxRelayRepository interface {
        FetchUnpublished(ctx context.Context, limit int) ([]model.Outbox, error)
        MarkPublished(ctx context.Context, id string) error
}

type KafkaProducer interface {
        Publish(ctx context.Context, key string, value interface{}) error
}

type OutboxRelay struct {
        repo     OutboxRelayRepository
        producer KafkaProducer
        Interval time.Duration
}

func NewOutboxRelay(repo OutboxRelayRepository, producer KafkaProducer) *OutboxRelay {
        return &OutboxRelay{
                repo:     repo,
                producer: producer,
                Interval: 1 * time.Second,
        }
}

func (r *OutboxRelay) Start(ctx context.Context) {
        go func() {
                ticker := time.NewTicker(r.Interval)
                defer ticker.Stop()

                for {
                        select {
                        case <-ctx.Done():
                                return
                        case <-ticker.C:
                                if err := r.poll(ctx); err != nil {
                                        log.Error().Err(err).Msg("outbox relay poll failed")
                                }
                        }
                }
        }()
}

func (r *OutboxRelay) poll(ctx context.Context) error {
        items, err := r.repo.FetchUnpublished(ctx, 100)
        if err != nil {
                return err
        }

        for _, item := range items {
                var payload map[string]interface{}
                if err := json.Unmarshal(item.Payload, &payload); err != nil {
                        log.Error().Err(err).Str("id", item.ID.String()).Msg("failed to unmarshal outbox payload")
                        continue
                }

                // Use payment_id or id as key for Kafka partitioning
                key := item.ID.String()
                if pID, ok := payload["payment_id"].(string); ok {
                        key = pID
                }

                if err := r.producer.Publish(ctx, key, payload); err != nil {
                        log.Error().Err(err).Str("id", item.ID.String()).Msg("failed to publish to Kafka")
                        // Exponential backoff or retry logic could be added here
                        // For now, we'll just skip and retry on next poll as per requirement
                        continue
                }

                if err := r.repo.MarkPublished(ctx, item.ID.String()); err != nil {
                        log.Error().Err(err).Str("id", item.ID.String()).Msg("failed to mark outbox as published")
                }
        }

        return nil
}

type RazorpayAPI interface {
        CreateOrder(ctx context.Context, amountPaise int64, currency, receipt string) (*RazorpayOrder, error)
        InitiateRefund(ctx context.Context, paymentID string, amountPaise int64) (*RazorpayRefund, error)
}

type CashfreeAPI interface {
        CreateOrder(ctx context.Context, amountPaise int64, currency, orderID string) (*CashfreeOrder, error)
}

type GatewayRouter struct {
        razorpay      RazorpayAPI
        cashfree      CashfreeAPI
        razorpayFails int64  // atomic counter
        circuitOpen   int32  // atomic bool (0=closed, 1=open)
        openedAt      int64  // unix timestamp
}

func NewGatewayRouter(razorpay RazorpayAPI, cashfree CashfreeAPI) *GatewayRouter {
        return &GatewayRouter{
                razorpay: razorpay,
                cashfree: cashfree,
        }
}

func (r *GatewayRouter) SelectGateway() (string, bool) {
        if atomic.LoadInt32(&r.circuitOpen) == 1 {
                openedAt := atomic.LoadInt64(&r.openedAt)
                if time.Now().Unix()-openedAt > 30 {
                        // Half-open: try Razorpay
                        return "RAZORPAY", true
                }
                // Circuit open: use Cashfree
                return "CASHFREE", false
        }
        return "RAZORPAY", false
}

func (r *GatewayRouter) RecordSuccess(gateway string) {
        if gateway == "RAZORPAY" {
                atomic.StoreInt64(&r.razorpayFails, 0)
                atomic.StoreInt32(&r.circuitOpen, 0)
        }
}

func (r *GatewayRouter) RecordFailure(gateway string) {
        if gateway == "RAZORPAY" {
                fails := atomic.AddInt64(&r.razorpayFails, 1)
                if fails >= 5 {
                        atomic.StoreInt32(&r.circuitOpen, 1)
                        atomic.StoreInt64(&r.openedAt, time.Now().Unix())
                        log.Warn().Msg("Razorpay circuit opened after 5 failures")
                }
        }
}

type PaymentService struct {
        repo          *repository.PaymentRepository
        gatewayRouter *GatewayRouter
}

func NewPaymentService(repo *repository.PaymentRepository, router *GatewayRouter) *PaymentService {
        return &PaymentService{
                repo:          repo,
                gatewayRouter: router,
        }
}

func (s *PaymentService) InitiatePayment(ctx context.Context, p *model.Payment) (*model.Payment, error) {
        // Check idempotency
        existing, err := s.repo.GetByIdempotencyKey(ctx, p.IdempotencyKey)
        if err != nil {
                return nil, err
        }
        if existing != nil {
                if existing.Status == model.PaymentStatusSuccess || existing.Status == model.PaymentStatusPending {
                        return existing, nil
                }
        }

        gatewayName, _ := s.gatewayRouter.SelectGateway()
        p.Gateway = gatewayName
        p.ID = uuid.New()

        if gatewayName == "RAZORPAY" {
                order, err := s.gatewayRouter.razorpay.CreateOrder(ctx, p.AmountPaise, p.Currency, p.ID.String())
                if err != nil {
                        s.gatewayRouter.RecordFailure("RAZORPAY")
                        return nil, fmt.Errorf("gateway error: %w", err)
                }
                s.gatewayRouter.RecordSuccess("RAZORPAY")
                p.GatewayOrderID = order.ID
        } else {
                // Cashfree fallback
                order, err := s.gatewayRouter.cashfree.CreateOrder(ctx, p.AmountPaise, p.Currency, p.ID.String())
                if err != nil {
                        return nil, fmt.Errorf("fallback gateway error: %w", err)
                }
                p.GatewayOrderID = order.OrderID
        }

        // Create outbox item
        payload, _ := json.Marshal(map[string]interface{}{
                "payment_id": p.ID,
                "status":     p.Status,
                "amount":     p.AmountPaise,
        })
        outboxItem := &model.Outbox{
                ID:    uuid.New(),
                Topic: "payments.initiated",
                Payload: payload,
        }

        if err := s.repo.CreateWithOutbox(ctx, p, outboxItem); err != nil {
                return nil, err
        }

        return p, nil
}

func (s *PaymentService) GetStatus(ctx context.Context, paymentID string) (*model.Payment, error) {
        return s.repo.GetByID(ctx, paymentID)
}

func (s *PaymentService) ListHistory(ctx context.Context, customerID string, limit, offset int) ([]model.Payment, error) {
        return s.repo.ListByCustomerID(ctx, customerID, limit, offset)
}

func (s *PaymentService) InitiateRefund(ctx context.Context, paymentID string, amountPaise int64) error {
        p, err := s.repo.GetByID(ctx, paymentID)
        if err != nil {
                return err
        }
        if p == nil {
                return fmt.Errorf("payment not found")
        }

        if p.Gateway == "RAZORPAY" {
                _, err := s.gatewayRouter.razorpay.InitiateRefund(ctx, p.GatewayPaymentID, amountPaise)
                if err != nil {
                        return err
                }
        } else {
                return fmt.Errorf("refund for %s not implemented", p.Gateway)
        }

        // Update status and outbox
        payload, _ := json.Marshal(map[string]interface{}{
                "payment_id": p.ID,
                "status":     model.PaymentStatusRefunded,
                "amount":     amountPaise,
        })
        outboxItem := &model.Outbox{
                ID:    uuid.New(),
                Topic: "payments.refunded",
                Payload: payload,
        }

        return s.repo.UpdateStatusWithOutbox(ctx, paymentID, model.PaymentStatusRefunded, p.GatewayPaymentID, "", "", outboxItem)
}
